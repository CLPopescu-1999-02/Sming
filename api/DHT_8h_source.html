00111 struct ComfortProfile
00112 {
00113     //Represent the 4 line equations:
00114     //dry, humid, hot, cold, using the y = mx + b formula
00115     float m_tooHot_m, m_tooHot_b;
00116     float m_tooCold_m, m_tooHCold_b;
00117     float m_tooDry_m, m_tooDry_b;
00118     float m_tooHumid_m, m_tooHumid_b;
00119 
00120     inline bool isTooHot(float temp, float humidity)
00121         {return (temp > (humidity * m_tooHot_m + m_tooHot_b));}
00122     inline bool isTooHumid(float temp, float humidity)
00123         {return (temp > (humidity * m_tooHumid_m + m_tooHumid_b));}
00124     inline bool isTooCold(float temp, float humidity)
00125         {return (temp < (humidity * m_tooCold_m + m_tooHCold_b));}
00126     inline bool isTooDry(float temp, float humidity)
00127         {return (temp < (humidity * m_tooDry_m + m_tooDry_b));}
00128 
00129     inline float distanceTooHot(float temp, float humidity)
00130         {return temp - (humidity * m_tooHot_m + m_tooHot_b);}
00131     inline float distanceTooHumid(float temp, float humidity)
00132         {return temp - (humidity * m_tooHumid_m + m_tooHumid_b);}
00133     inline float distanceTooCold(float temp, float humidity)
00134         {return (humidity * m_tooCold_m + m_tooHCold_b) - temp;}
00135     inline float distanceTooDry(float temp, float humidity)
00136         {return (humidity * m_tooDry_m + m_tooDry_b) - temp;}
00137 };
